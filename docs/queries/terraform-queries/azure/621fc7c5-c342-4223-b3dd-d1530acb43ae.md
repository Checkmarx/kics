### Description
All 'azurerm_storage_account' resources should use the latest SMB protocol version to prevent exploitation of known vulnerabilities<br>
[Documentation](https://registry.terraform.io/providers/hashicorp/azurerm/latest/docs/resources/storage_account#smb-2)

### Code samples
#### Code samples with security vulnerabilities
```tf title="Positive test num. 1 - tf file" hl_lines="32 1 75 47 18 61"
resource "azurerm_storage_account" "positive1" {
  name                     = "positive1"
  resource_group_name      = azurerm_resource_group.positive1.name
  location                 = azurerm_resource_group.positive1.location
  account_tier             = "Standard"
  account_replication_type = "GRS"

  # missing "share_properties" (allows all SMB protocols)
}

resource "azurerm_storage_account" "positive2" {
  name                     = "positive2"

  location                 = azurerm_resource_group.positive2.location

  account_replication_type = "GRS"

    # missing "smb" (allows all SMB protocols)
  }

resource "azurerm_storage_account" "positive3" {
```
