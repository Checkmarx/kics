### Description
All 'azurerm_storage_account' resources should exclusively use 'AES-256-GCM' for channel encryption<br>
[Documentation](https://registry.terraform.io/providers/hashicorp/azurerm/latest/docs/resources/storage_account#channel_encryption_type-2)

### Code samples
#### Code samples with security vulnerabilities
```tf title="Positive test num. 1 - tf file" hl_lines="1 74 46 18 60 31"
resource "azurerm_storage_account" "positive1" {
  name                     = "positive1"
  resource_group_name      = azurerm_resource_group.positive1.name
  location                 = azurerm_resource_group.positive1.location
  account_tier             = "Standard"
  account_replication_type = "GRS"

  # missing "share_properties" (allows all encryption standards)
}

resource "azurerm_storage_account" "positive2" {

  resource_group_name      = azurerm_resource_group.positive2.name

  account_replication_type = "GRS"

resource "azurerm_storage_account" "positive3" {
```
